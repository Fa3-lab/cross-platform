Solution 1: Arrow Functions ‚Üí Regular Functions

Replace arrow functions with traditional function expressions

‚ùå PROBLEM: Modern ES6 Code
// Basic arrow function const add = (a, b) => a + b; // Array methods with arrows const numbers = [1, 2, 3, 4, 5]; const doubled = numbers.map(n => n * 2); const evens = numbers.filter(n => n % 2 === 0); // Lexical 'this' binding class Timer { constructor() { this.seconds = 0; setInterval(() => { this.seconds++; // 'this' refers to Timer instance console.log(this.seconds); }, 1000); } }
‚úÖ SOLUTION: Cross-Browser Compatible

String interpolation and multiline strings

ES6 Template Literals
// String interpolation const name = 'World'; const greeting = `Hello, ${name}!`; // Expressions in templates const a = 5, b = 10; const result = `The sum of ${a} and ${b} is ${a + b}`; // Multiline strings const multiline = ` This is a multiline string with proper indentation `; // Tagged templates function highlight(strings, ...values) { return strings.reduce((result, string, i) => { const value = values[i] ? `${values[i]}` : ''; return result + string + value; }, ''); } const highlighted = highlight`Hello ${name}, you have ${5} messages`;

// String concatenation var name = 'World'; var greeting = 'Hello, ' + name + '!'; // Expressions require concatenation var a = 5, b = 10; var result = 'The sum of ' + a + ' and ' + b + ' is ' + (a + b); // Multiline strings (awkward) var multiline = '\n' + ' This is a\n' + ' multiline string\n' + ' with manual line breaks\n'; // No equivalent to tagged templates function highlight(template, name, count) { return template .replace('${name}', '' + name + '') .replace('${count}', '' + count + ''); } var highlighted = highlight('Hello ${name}, you have ${count} messages', name, 5);

// Array methods polyfills if (!Array.prototype.includes) { Array.prototype.includes = function(item) { return this.indexOf(item) !== -1; }; } if (!Array.prototype.find) { Array.prototype.find = function(callback) { for (var i = 0; i < this.length; i++) { if (callback(this[i], i, this)) { return this[i]; } } return undefined; }; } // Promise polyfill (simplified) if (typeof Promise === 'undefined') { window.Promise = function(executor) { var self = this; self.state = 'pending'; self.value = undefined; self.handlers = []; function resolve(result) { if (self.state === 'pending') { self.state = 'fulfilled'; self.value = result; self.handlers.forEach(handle); } } function reject(error) { if (self.state === 'pending') { self.state = 'rejected'; self.value = error; self.handlers.forEach(handle); } } executor(resolve, reject); }; } // Object.assign polyfill if (typeof Object.assign !== 'function') { Object.assign = function(target) { for (var i = 1; i < arguments.length; i++) { var source = arguments[i]; for (var key in source) { if (source.hasOwnProperty(key)) { target[key] = source[key]; } } } return target; }; }
