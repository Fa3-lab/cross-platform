// Feature detection and fallback validation function validateForm() { const form = document.getElementById('validation-form'); // Check if HTML5 validation is supported if (typeof form.checkValidity === 'function') { // Use native HTML5 validation return form.checkValidity(); } else { // Fallback to manual validation return manualValidation(form); } } function manualValidation(form) { let isValid = true; const inputs = form.querySelectorAll('input[required]'); inputs.forEach(input => { if (!input.value.trim()) { showError(input, 'This field is required'); isValid = false; } }); return isValid; }
